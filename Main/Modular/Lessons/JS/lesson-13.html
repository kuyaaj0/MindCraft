let scores = [10, 20, 30, 40];

YOUR MISSION: Inventory Manager!

Lesson 13 ‚Ä¢ Topic 5
Complete inventory system using ALL array methods! 
let inventory = ["sword", "potion", "shield"];
let values = [200, 50, 150];
let enemies = ["goblin", "orc", "dragon", "boss"];

// Add new items
inventory.push("bow", "ring");
console.log("Updated inventory:", inventory);

// Remove used potion
let usedItem = inventory.shift();
console.log("Used:", usedItem);

// Get valuable items (>100)
let valuables = inventory.filter(item => item.length > 4);
console.log("Valuables:", valuables);

// Calculate total value (simplified)
let itemValues = values.map(v => v * 2);
let totalValue = itemValues.reduce((sum, val) => sum + val, 0);
console.log("Total value:", totalValue);

// Sort enemies by name
enemies.sort();
console.log("Enemy order:", enemies);

// Find boss
let boss = enemies.find(e => e === "boss");
console.log("Target:", boss);

// Print inventory
inventory.forEach((item, i) => console.log(`${i+1}. ${item}`));
