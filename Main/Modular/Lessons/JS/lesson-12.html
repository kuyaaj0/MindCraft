Lesson 12 ‚Ä¢ Topic 5
Create random game values: loot, enemies, damage! 
function randomRange(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

console.log("Enemy level:", randomRange(1, 10));
console.log("Loot coins:", randomRange(5, 50));
console.log("Damage roll:", randomRange(10, 25));
console.log("Rare drop %:", Math.floor(Math.random() * 100) + 1);
