01. What is Local Storage?

localStorage allows websites to store small pieces of data directly in the user's browser. It‚Äôs permanent (until manually cleared) and perfect for saving notes, preferences, or app state ‚Äì all without a database.

02. Storing and Retrieving Data

Use these commands to save and load values:
  • localStorage.setItem("key", "value") ‚Üí Save data.
  • localStorage.getItem("key") ‚Üí Retrieve data.
  • localStorage.removeItem("key") ‚Üí Delete one entry.
  • localStorage.clear() ‚Üí Clear all data.
<script>
localStorage.setItem("username", "Neo");
let user = localStorage.getItem("username");
console.log("Welcome back,", user);
</script>
Output ‚Üí Logs ‚ÄúWelcome back, Neo‚Äù to the console.

03. Building a Cyber Notes App

Let‚Äôs create an app that saves notes directly into localStorage. Even if you close the browser, your data stays safe inside the neon matrix! 
    <script>
const input = document.getElementById("noteInput");
const addBtn = document.getElementById("addNote");
const clearBtn = document.getElementById("clearAll");
const list = document.getElementById("noteList");

// Load saved notes
let notes = JSON.parse(localStorage.getItem("notes")) || [];
renderNotes();

addBtn.addEventListener("click", () => {
  if(input.value.trim() === "") return;
  notes.push(input.value);
  localStorage.setItem("notes", JSON.stringify(notes));
  input.value = "";
  renderNotes();
});

clearBtn.addEventListener("click", () => {
  notes = [];
  localStorage.clear();
  renderNotes();
});

function renderNotes(){
  list.innerHTML = "";
  notes.forEach((note, index) => {
    let li = document.createElement("li");
    li.textContent = note;
    let del = document.createElement("button");
    del.textContent = "Delete";
    del.onclick = () => { notes.splice(index,1);
      localStorage.setItem("notes", JSON.stringify(notes));
      renderNotes();
    };
    li.appendChild(del);
    list.appendChild(li);
  });
}
</script>
    Output ‚Üí Type a note, click ‚ÄúAdd Note‚Äù, and it stays even after refreshing!
