01. Numbers - Math Power!

Numbers = math data (whole, decimals). No quotes! JavaScript handles integers + decimals automatically. 
let age = 20;              // Integer (whole number)
let score = 95.5;          // Decimal (float)
let pi = 3.14159;          // Math constant

console.log(10 + 5);       // 15 (Addition)
console.log(10 - 5);       // 5  (Subtraction) 
console.log(10 * 5);       // 50 (Multiplication)
console.log(10 / 5);       // 2  (Division)

02. Strings - Text Power!

Strings = text data. Always use "" or '' quotes! 
let name = "Jhack";        // Double quotes ""
let course = 'BSIT';       // Single quotes ''
let message = `Hello ${name}`; // Template literals `` (backticks)

console.log(name + " " + course);  // Concatenation
console.log(`Score: ${95}`);      // Template literal

03. Booleans - TRUE/FALSE Power!

Booleans = true/false logic. Used for conditions, flags, decisions. 
let isActive = true;       // TRUE = on
let isGraduated = false;   // FALSE = off

let score = 95;
let passed = score >= 75;  // Comparison ‚Üí Boolean

console.log(passed);       // true

04. Special Types - null & undefined

null = "empty on purpose". undefined = "never set".
  • let user = null; // Empty user (deleted)
    let score; // undefined (never assigned) 
let futureStudent = null;      // No student assigned yet
let missingScore;              // Never set = undefined

console.log(futureStudent);    // null
console.log(missingScore);     // undefined

05. Assignment & Math Operators

= assigns values. Math operators +-*/ do calculations. 
// ARITHMETIC OPERATORS
let score = 85;        // =
score = score + 10;    // 95
score += 5;            // 100 (shorthand)
score++;               // 101 (+1)
score--;               // 100 (-1)

// COMPARISON OPERATORS
let passed = score >= 75;  // true
let elite = score === 100; // exact match

100 ‚Üí true ‚Üí false
